6. Crear un procedimiento en pseudocodigo que devuelva la cantidad de n√∫meros primos que hubo en un conjunto de n numero ingresados por teclado

boolean esprimo (int a) {
	c <- 0
	for int i <- 2 to a / 2 + 1 step 1 {
		if (a MOD i = 0) {
			c <- c + 1
		}
	}
	if (c = 0) {
		return true
	}
	else {
		return false
	}
}
Main {
	int n, x, c <- 0
	Read (x)
	for int i <- 0 to n step 1 {
		Read (n)
		if (esprimo(n) = true) {
			c <- c + 1
		}
	}
	Print ("La cantidad de numeros primos es: " , c)
}
